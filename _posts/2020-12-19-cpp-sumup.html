指针

#include <iostream>
using namespace std;

int main()
{
  int a = 123;
  int *b = &a;
  int **c = &b;
  cout<<a<<endl; //123
  cout<<b<<endl; //address of a
  cout<<c<<endl; //address of b
  cout<<*b<<endl; // 123
  cout<<*c<<endl; //address of b
  cout<<**c<<endl; // 123
  return 0;
}

const 指针

const * 与 * const区别 

int a = 5;
int b = 6;
const int * p1 = &a;
p1 = &b; //可行，可以这样赋值
a = 55; //错误，p2指向的地址内的内容不可改变

int c = 7;
int d = 8;
int * const p2 = &c;
p2 = &d; //错误，p2指向地址不可以改变
c = 77; //可以

总结：const在*前面，指针指向内容不可改变。（可以理解为int const (*p1)，const的是*p1这个数值）。 const在*后面，指针指向地址不可改变。（可以理解为int * const(p1)，const的是p1这个地址）。
